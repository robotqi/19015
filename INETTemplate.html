010public interface INETTemplate
011{
012    /**
013     * Accept a full file path and name and determine if it exists on disk. If it does, return a true. If it does not exist, return a false.
014     * @param strFileName
015     * @return
016     */
017    public Boolean fileExists(String strFileName);
018
019    /**
020     * Accept a full file path and name as a parameter and check to see if it exists using the fileExists method. If it does exist, read the file from disk and return the contents as a string. If the file does not exist, return a null.
021     * @param strFileName
022     * @return
023     * @throws Exception
024     */
025    public String getFromFile(String strFileName) throws Exception;
026
027    /**
028     * Accept a full file path and name as a parameter and check to see if it exists using the fileExists method. If it does exist, read the file from disk and return the contents as a string. If the file does not exist, return a null.
029     * @param strPage
030     * @return
031     */
032    public String getPREData(String strPage);
033
034    /**
035     * Extract a small string from a larger one using a Regular Expression and the pattern specified as a parameter. If nothing is found, return a null.
036     * @param strInput
037     * @param strPatten
038     * @return
039     */
040    public String getRegEx(String strInput, String strPatten);
041
042    /**
043     * Accept a string URL, create a string using the StringBuilder class and return it as one long string. If the page does not exist, return a null.
044     * @param strURL
045     * @return
046     * @throws Exception
047     */
048    public String getURLRaw(String strURL) throws Exception;
049
050    /**
051     * Accept a string and convert the whole string into lower case then upper case only the first letter of each word.
052     * @param strInput
053     * @return
054     */
055    public String properCase(String strInput);
056
057    /**
058     * Accept a string and save it to a local file.
059     * @param strFileName
060     * @param strContent
061     * @throws Exception
062     */
063    public void saveToFile(String strFileName, String strContent) throws Exception;
064
065}

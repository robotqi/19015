016public interface XMLTemplate
017{
018    int intRecordPos = -1;
019    /**
020     This function accepts a URL to an XML page and stores it in a
021     Document object.  The class variable intRecNum is also set to -1.
022     @param strURL - URL to the XML page (can be local or over the Internet)
023     */
024    public void loadPage(String strURL) throws Exception;
025    /**
026     This function looks for a table name in the XML document
027     and returns the first child node.
028     @param strTable - the name of the root table
029     @return string - returns the child node as a string
030     */
031    public String setTable(String strTable) throws Exception;
032    /**
033     This function will search for a value represented by strKey in
034     a field indicated by strKeyField.
035
036     @param strKeyName - the field name to search
037     @param strKeyContents - the value of the field name to be searched
038     @return integer - the record number(position) or a minus one if it did not find the key
039     */
040    public int find(String strKeyName, String strKeyContents) throws Exception;
041    /**
042     This method extracts the contents of the field at the
043     the CURRENT position as determined by intRecNum.
044
045     @param strFieldName - the field name of the data you want to retrieve
046     @return integer - the contents of the field
047     */
048    public String getField(String strFieldName) throws Exception;
049    /**
050     This method sets the current record to intRec and then
051     extracts the contents of that field at that record.
052     @param strField - the field name of the data you want to retrieve
053     @param intRec - the record number to use
054     @return string - the contents of the field
055     */
056    public String getField(String strField, int intRec) throws Exception;
057    /**
058     This method returns the number of records in the XML
059     file after a getElementsByTagName() search
060     @param strFieldName - this can be any field that is in the record
061     @return integer - the number of records in that table
062     */
063    public int getRecordCount(String strFieldName);
064    /**
065     This method is just used to display data during testing.
066     @param strVar - any string (it is displayed in the jGrasp messages window)
067     */
068    public void status(String strVar);
069    /**
070     Returns a string about this object.
071     */
072    public String toString();
073
074}

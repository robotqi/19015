023public class XMLRead implements XMLTemplate
024{
025    /** Once the XML file is read, it is stored in this object. */
026    private Document document;
027    /** This is a class level variable that maintains the current record (node)
028     number in use.  It is set by find()and used by getField().
029     */
030    private int intRecNum;
031    /** The constructor only initializes intRecNum equal to a minus one. */
032    public XMLRead()
033    {
034        intRecNum = -1;
035    }
036    /**
037     This function accepts a URL to an XML page and stores it in a
038     Document object.  The class variable intRecNum is also set to -1.
039     */
040    public void loadPage(String strURL) throws Exception
041    {
042        String strFileName="";
043        String strContent="";
044        status("readXML: URL=" + strURL);
045
046        URL myWebAddress= new URL(strURL);
047        status("readXML: URL is set up");
048        URLConnection myConnection = myWebAddress.openConnection();
049        status("readXML: Connection opened");
050        InputStream myStream = myConnection.getInputStream();
051        status("readXML: Get URL done");
052
053        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
054        status("readXML: Factory built");
055        DocumentBuilder builder = factory.newDocumentBuilder();
056        status("readXML: Builder built");
057        document = builder.parse(myStream);
058        status("readXML: Document is done");
059        intRecNum = 0;
060    }
061    /**
062     This function looks for a table name in the XML document
063     and returns the first child node.
064     */
065    public String setTable(String strTable) throws Exception
066    {
067        NodeList list = document.getElementsByTagName(strTable);
068        Node node = list.item(0);
069        TransformerFactory tFactory = TransformerFactory.newInstance();
070        Transformer transformer = tFactory.newTransformer();
071        DOMSource source = new DOMSource(node);
072        return source.getNode().getFirstChild().toString();
073    }
074    /**
075     This function will search for a value represented by strKey in
076     a field indicated by strKeyField.
077     <ul>
078     <li>intRecNum is set to a minus one if it did not find the key</li>
079     <li>intRecNum is set to the index number (position) if it did find the key.
080     </ul>
081     */
082    public int find(String strKeyName, String strKeyContents) throws Exception
083    {
084        intRecNum = -1;  // Class Level Variable
085        // create an array of nodes of strKeyFields
086        NodeList list = document.getElementsByTagName(strKeyName);
087        // start searching at the beginning
088        int intCount = 0;
089        // flag to stop the search once found
090        boolean blnContinue = true;
091        while (intCount < list.getLength() && blnContinue)
092        {
093            // list.item contains an array of nodes made up of Fields.
094            // getFirstChild() gets the first child node (which is the record itself)
095            // getNodeValue() gets the text contents of that field
096            if (list.item(intCount).getFirstChild().getNodeValue().equals(strKeyContents))
097            {
098                intRecNum = intCount;
099                blnContinue = false;
100            }
101            intCount++;
102        }
103        return intRecNum;
104    }
105    /**
106     This method extracts the contents of the field at the
107     the current intRecNum position.
108     */
109    public String getField(String strFieldName) throws Exception
110    {
111        String strRet="";
112        NodeList list = document.getElementsByTagName(strFieldName);
113        if (intRecNum > -1 && intRecNum < list.getLength())
114        {
115            strRet = list.item(intRecNum).getFirstChild().getNodeValue().trim();
116        }
117        return strRet;
118    }
119    /**
120     This method sets the current record to intRec and then
121     extracts the contents of that field at that record.
122     */
123    public String getField(String strField, int intRec) throws Exception
124    {
125        if (intRec > -1)
126        {
127            intRecNum = intRec; // Class Variable
128            return getField(strField);
129        }
130        else
131        {
132            return "";
133        }
134    }
135    /**
136     This method returns the number of records in the XML
137     file after a getElementsByTagName() search
138     */
139    public int getRecordCount(String strFieldName)
140    {
141        NodeList list = document.getElementsByTagName(strFieldName);
142        return list.getLength();
143    }
144    /**
145     This method is just used to display data during testing.
146     */
147    public void status(String strVar)
148    {
149        System.out.println("XMLRead: " + strVar);
150    }
151}
